    function explanationForCurrentStep() {
      const step = steps[Math.max(0, state.activeStep)];
      if (!step) return 'Uruchom symulację, aby zobaczyć opis aktywnego etapu.';
      const cacheNote = refs.cacheToggle.checked ? ' Cache skraca część etapów, ale przeglądarka nadal musi upewnić się, że może użyć lokalnych danych.' : '';
      return `${step.title}: ${step.description}${cacheNote}`;
    }

    function renderDetails() {
      refs.codeBox.textContent = codeForCurrentTab();
      refs.explainBox.textContent = explanationForCurrentStep();
    }

    function renderBrowserPreview(final = false) {
      refs.addressBar.textContent = refs.urlInput.value.trim() || 'https://example.edu/kurs/web';
      if (!final) {
        refs.renderedPage.innerHTML = `
          

          

          

          

        `;
        return;
      }
      refs.renderedPage.innerHTML = `
        

          
Strona wyrenderowana

          
Przeglądarka pobrała dokument, zbudowała DOM i CSSOM, policzyła layout oraz namalowała piksele. Interfejs jest gotowy do interakcji.

        

        

        

        

      `;
    }

    function addLog(message) {
      const item = document.createElement('p');
      const elapsed = String(Math.round(state.totalTime)).padStart(4, ' ');
      item.textContent = `[${elapsed} ms] ${message}`;
      refs.log.prepend(item);
    }

    function updateNodeState() {
      const active = steps[state.activeStep]?.id || 'idle';
      refs.clientState.textContent = state.running ? 'pracuje' : 'idle';
      refs.networkState.textContent = state.running ? networkProfiles[refs.networkSelect.value].label : 'czeka';
      refs.serverState.textContent = state.activeStep >= 4 ? '200' : 'gotowy';
      refs.packetLabel.textContent = state.running ? active.toUpperCase() : 'brak pakietu';
      refs.serverScreen.textContent = refs.cacheToggle.checked && state.activeStep >= 4
        ? 'Część zasobów wraca z cache, serwer obsługuje tylko potrzebne odświeżenia.'
        : 'HTML, CSS, JS i obrazy są gotowe do wysłania.';
    }

    function updateMetrics() {
      refs.totalTime.textContent = formatMs(state.totalTime);
      refs.resourceCount.textContent = `${state.resourceDone}/4`;
      refs.statusText.textContent = state.running ? 'Trwa' : state.completed.size === steps.length ? 'Gotowe' : 'Gotowe';
      refs.latencyValue.textContent = `${refs.latencyRange.value} ms`;
    }

    function renderAll() {
      renderPipeline();
      renderWaterfall();
      renderResources();
      renderDetails();
      updateNodeState();
      updateMetrics();
    }

    function resetSimulation() {
      state.activeStep = -1;
      state.completed = new Set();
      state.totalTime = 0;
      state.resourceDone = 0;
      state.running = false;
      refs.log.innerHTML = '';
      renderBrowserPreview(false);
      renderAll();
    }

    function wait(ms) {
      return new Promise((resolve) => window.setTimeout(resolve, Math.min(ms, 650)));
    }

    async function runSimulation() {
      if (state.running) return;
      resetSimulation();
      state.running = true;
      addLog('Start nawigacji do adresu URL.');
      renderAll();
      for (let index = 0; index < steps.length; index += 1) {
        state.activeStep = index;
        const step = steps[index];
        const stepTime = getStepTime(step);
        addLog(`${step.title}: start.`);
        if (step.id === 'response') state.resourceDone = refs.cacheToggle.checked ? 3 : 1;
        if (step.id === 'parse') state.resourceDone = refs.cacheToggle.checked ? 4 : 3;
        if (step.id === 'paint') state.resourceDone = 4;
        renderAll();
        await wait(stepTime);
        state.totalTime += stepTime;
        state.completed.add(index);
        addLog(`${step.title}: zakończono po ${formatMs(stepTime)}.`);
        renderAll();
      }
      state.activeStep = steps.length - 1;
      state.running = false;
      renderBrowserPreview(true);
      addLog('Pierwszy render zakończony. Strona jest widoczna.');
      renderAll();
    }

    document.getElementById('runButton').addEventListener('click', runSimulation);
    document.getElementById('resetButton').addEventListener('click', resetSimulation);

    [refs.urlInput, refs.latencyRange, refs.networkSelect, refs.cacheToggle, refs.compressionToggle, refs.blockingToggle].forEach((control) => {
      control.addEventListener('input', () => {
        refs.addressBar.textContent = refs.urlInput.value.trim() || 'https://example.edu/kurs/web';
        renderAll();
      });
      control.addEventListener('change', renderAll);
    });

    document.querySelectorAll('.tab').forEach((tab) => {
      tab.addEventListener('click', () => {
        document.querySelectorAll('.tab').forEach((item) => item.classList.remove('active'));
        tab.classList.add('active');
        state.currentTab = tab.dataset.tab;
        renderDetails();
      });
    });

    document.querySelectorAll('.scenario').forEach((button) => {
      button.addEventListener('click', () => {
        document.querySelectorAll('.scenario').forEach((item) => item.classList.remove('active'));
        button.classList.add('active');
        state.scenario = button.dataset.scenario;
        if (state.scenario === 'cached') {
          refs.cacheToggle.checked = true;
          refs.networkSelect.value = 'fast';
          refs.latencyRange.value = 80;
        } else if (state.scenario === 'slowcss') {
          refs.cacheToggle.checked = false;
          refs.networkSelect.value = 'slow';
          refs.latencyRange.value = 320;
          refs.blockingToggle.checked = true;
        } else {
          refs.cacheToggle.checked = false;
          refs.networkSelect.value = 'fast';
          refs.latencyRange.value = 120;
          refs.blockingToggle.checked = true;
        }
        resetSimulation();
      });
    });

    document.querySelectorAll('.quiz button').forEach((button) => {
      button.addEventListener('click', () => {
        document.querySelectorAll('.quiz button').forEach((item) => item.classList.remove('correct', 'incorrect'));
        button.classList.add(button.dataset.answer === 'correct' ? 'correct' : 'incorrect');
      });
    });
